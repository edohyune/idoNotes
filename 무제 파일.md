public static T CloneWithHistory<T>(T source) where T : new()
{
    T clone = new T();
    foreach (var prop in typeof(T).GetProperties())
    {
        if (prop.CanRead && prop.CanWrite)
        {
            prop.SetValue(clone, prop.GetValue(source));
            
            // '_old' 프로퍼티가 있다면, 그 값도 설정
            var historyProp = typeof(T).GetProperty($"{prop.Name}_old");
            if (historyProp != null && historyProp.CanWrite)
            {
                historyProp.SetValue(clone, prop.GetValue(source));
            }
        }
    }
    return clone;
}
